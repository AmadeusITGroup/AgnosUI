Props

allowedPlacements

Type: Placement[]
List of allowed placements for the dropdown. This refers to the [allowedPlacements from floating UI](https://floating-ui.com/docs/autoPlacement#allowedplacements), given the different [Placement possibilities](https://floating-ui.com/docs/computePosition#placement).
Default value: ['bottom-start', 'top-start', 'bottom-end', 'top-end']

ariaLabel

Type: string | undefined
aria-label used for the input inside the select
Default value: 'Select'

badgeClassName

Type: string
Class to be added on selected items (displayed in the input zone)
Default value: ''

className

Type: string
CSS classes to be applied on the widget main container
Default value: ''

disabled

Type: boolean
true if the select is disabled
Default value: false

filterText

Type: string
Filtered text to be display in the filter input
Default value: ''

id

Type: string | undefined
id used for the input inside the select
Default value: undefined

itemIdFn

Type: (item: T) => string
Custom function to get the id of an item By default, the item is returned
Default value: 
defaultItemId

items

Type: T[]
List of available items for the dropdown
Default value: []

loading

Type: boolean
true if a loading process is being done
Default value: false

menuClassName

Type: string
Class to be added on the dropdown menu container
Default value: ''

menuItemClassName

Type: string
Class to be added on menu items
Default value: ''

navSelector

Type: (node: HTMLElement) => NodeListOf<HTMLSpanElement | HTMLInputElement>
Retrieves navigable elements within an HTML element containing badges and the input.
Default value: 
(node: HTMLElement) => node.querySelectorAll('.au-select-badge,input')

onFilterTextChange

Type: (text: string) => void
Callback called when the text filter change
Default value: 
noop

onOpenChange

Type: (isOpen: boolean) => void
Callback called dropdown open state change
Default value: 
noop

onSelectedChange

Type: (selected: T[]) => void
Callback called when the selection change
Default value: 
noop

open

Type: boolean
true if the select is open
Default value: false

selected

Type: Item[]
List of selected item ids
Default value: []

slotBadgeLabel

Type: SlotContent<SelectItemContext<Item>>
The template to override the way each badge on the left of the input is displayed. This define the content of the badge inside the badge container.
Default value: ({itemContext}) => itemContext.item

slotItem

Type: SlotContent<SelectItemContext<Item>>
The template to override the way each item is displayed in the list. This define the content of the badge inside the badge container.
Default value: ({itemContext}) => itemContext.item

Api

clear

Type: () => void
Clear all the selected items

clearText

Type: () => void
Clear the filter text

close

Type: () => void
close the select

highlight

Type: (item: Item) => void
Highlight the given item, if there is a corresponding match among the visible list

highlightFirst

Type: () => void
Highlight the first item among the visible list

highlightLast

Type: () => void
Highlight the last item among the visible list

highlightNext

Type: () => void
Highlight the next item among the visible list. Loop to the first item if needed

highlightPrevious

Type: () => void
Highlight the previous item among the visible list Loop to the last item if needed

open

Type: () => void
open the select

select

Type: (item: Item) => void
Select the provided item. The selected list is used to

toggle

Type: (isOpen?: boolean) => void
Toggle the dropdown menu

toggleItem

Type: (item: Item, selected?: boolean) => void
Toggle the selection of an item

unselect

Type: (item: Item) => void
Unselect the provided item.
